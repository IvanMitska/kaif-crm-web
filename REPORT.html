Команды разработки

# Запуск
pnpm dev # Все сервисы
pnpm backend:dev # Только backend
pnpm frontend:dev # Только frontend

# Docker
pnpm docker:up # Поднять контейнеры
pnpm docker:down # Остановить

# База данных
pnpm db:migrate # Миграции
pnpm db:seed # Сидирование
